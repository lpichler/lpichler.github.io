200 def users_from_proxy(self, user, query_params, options, limit, offset): 

201 """Format principal request for proxy and return prepped result.""" 

202 proxy = PrincipalProxy() 

203 usernames = query_params.get(USERNAMES_KEY) 

204 email = query_params.get(EMAIL_KEY) 

205 match_criteria = validate_and_get_key(query_params, MATCH_CRITERIA_KEY, VALID_MATCH_VALUE, "exact") 

206 options["username_only"] = validate_and_get_key(query_params, USERNAME_ONLY_KEY, VALID_BOOLEAN_VALUE, "false")

208 if not usernames and not email: 

209 options["admin_only"] = validate_and_get_key(query_params, ADMIN_ONLY_KEY, VALID_BOOLEAN_VALUE, "false") 

210 resp = proxy.request_principals(org_id=user.org_id, limit=limit, offset=offset, options=options)
