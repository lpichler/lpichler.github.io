108 def find_group_with_tuples( 

109 self, 

110 predicates: Union[Callable[[RelationTuple], bool], List[Callable[[RelationTuple], bool]]], 

111 group_by: Callable[[RelationTuple], T], 

112 group_filter: Callable[[T], bool] = lambda _: True, 

113 require_full_match: bool = False, 

114 match_once: bool = True, 

115 ) -> Tuple[dict[T, List[RelationTuple]], dict[T, List[RelationTuple]]]: 

116 """ 

117 Find groups of tuples matching given predicates, grouped by a key. 

118 

119 Groups the tuples using the provided `group_by` function and tests the 

120 predicates against each group independently. 

121 

122 For each group, this method attempts to find tuples that match all the 

123 predicates, ensuring that no tuple is used for more than one predicate 

124 within the group. 

125 

126 If `require_full_match` is True, the method also ensures that all tuples 

127 in the group are matched by the predicates (i.e., no unmatched tuples 

128 remain in the group). If any group does not meet the criteria, it is 

129 excluded from the results. 

130 

131 Args: 

132 predicates: A list of predicates (functions) that each accept a 

133 RelationTuple and return a bool indicating a match. 

134 group_by: A function that takes a RelationTuple and returns a key 

135 to group by (e.g., a resource ID). 

136 require_full_match: If True, only groups where all tuples are matched 

137 by the predicates (i.e. there are no remaining unmatched tuples) 

138 are included in the results. 

139 group_filter: A predicate that filters the groups to include in the 

140 results. Useful when you only want to test a subset of tuples e.g. 

141 a specific resource type. 

142 match_once: If True, each predicate is only used once in the matching process. 

143 Otherwise, each tuple in the group will be tested by each predicate until 

144 one predicate matches the tuple. 

145 

146 Returns: 

147 A tuple containing two dictionaries: 

148 - The first dictionary contains the groups that matched all predicates. 

149 - The second dictionary contains the groups that did not match all predicates. 

150 """ 

151 # Group the tuples by the specified key 

152 grouped_tuples: dict[T, List[RelationTuple]] = defaultdict(list) 

153 for rel in self._tuples: 

154 key = group_by(rel) 

155 if group_filter(key): 

156 grouped_tuples[key].append(rel) 

157 

158 matching_groups: dict[T, List[RelationTuple]] = {} 

159 unmatched_groups: dict[T, List[RelationTuple]] = {} 

160 

161 # Iterate over each group 

162 predicate_list = predicates if isinstance(predicates, list) else [predicates] 

163 for key, group_tuples in grouped_tuples.items(): 

164 remaining_tuples = list(group_tuples) 

165 remaining_predicates = list(predicate_list) if match_once else predicate_list 

166 i = 0 

167 matching_tuples = [] 

168 success = True 

169 

170 # Attempt to match all predicates within the group 

171 # Using each predicate only once if requested 

172 while remaining_predicates and i < len(remaining_predicates): 

173 predicate = remaining_predicates[i] 

174 found = False 

175 j = 0 

176 while j < len(remaining_tuples): 

177 rel = remaining_tuples[j] 

178 if predicate(rel): 

179 matching_tuples.append(rel) 

180 remaining_tuples.pop(j) 

181 found = True 

182 if match_once: 

183 break # Move to next predicate 

184 else: 

185 j += 1 

186 if not found: 

187 success = False 

188 break # Predicate not satisfied in this group 

189 if match_once: 

190 remaining_predicates.pop(i) 

191 else: 

192 i += 1 

193 

194 if require_full_match and remaining_tuples or not success: 

195 unmatched_groups[key] = group_tuples 

196 continue 

197 

198 matching_groups[key] = matching_tuples 

199 

200 return matching_groups, unmatched_groups
